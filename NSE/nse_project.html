my_files <- list.files(path = "data/", 
                       pattern = "stocks_\\d{4}\\.csv$", 
                       full.names = TRUE, 
                       recursive = TRUE)



full_data <- map_dfr(my_files, read_csv) |>
  janitor::clean_names() |>
  mutate(date = lubridate::dmy(date))
